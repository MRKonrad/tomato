259  template< typename MeasureType >
260  int
261  CalculatorT1WithSignCheck<MeasureType>
262  ::calculateInvCovarianceMatrix(const MeasureType* invTimes, const MeasureType* residuals, const MeasureType* parameters, MeasureType *invCovarianceMatrix) {
263 
264  // invCovarianceMatrix - indexing by column,row
265 
266  int nSamples = this->getNSamples();
267  MeasureType A = parameters[0];
268  MeasureType B = parameters[1];
269  MeasureType T1star = parameters[2];
270  MeasureType tempInvCovarianceMatrix[3*3];
271 
272  for (int i = 0; i < 3*3; ++i) {
273  invCovarianceMatrix[i] = 0;
274  tempInvCovarianceMatrix[i] = 0;
275  }
276 
277  if (fabs(A) < std::numeric_limits<MeasureType>::min())
278  return 1; // EXIT_FAILURE
279  if (fabs(T1star) < std::numeric_limits<MeasureType>::min())
280  return 1; // EXIT_FAILURE
281 
282  MeasureType T1 = (B/A-1)*T1star;
283 
284  if (fabs(T1) < std::numeric_limits<MeasureType>::min())
285  return 1; // EXIT_FAILURE
286 
287  MeasureType dydA = 0;
288  MeasureType dydB = 0;
289  MeasureType dydT1 = 0;
290 
291  for (int i = 0; i < nSamples; ++i){
292  MeasureType invTime = invTimes[i];
293  MeasureType myexparg = ( -invTime * ( B/A - 1) / T1);
294 
295  if ((myexparg > std::numeric_limits<MeasureType>::max_exponent)
296  || (myexparg < std::numeric_limits<MeasureType>::min_exponent))
297  return 1; //EXIT_FAILURE
298 
299  MeasureType myexp = exp (myexparg);
300  dydA = 1 - B * myexp * invTime * B / ( T1 * A * A);
301  dydB = -myexp + B * myexp * invTime / ( T1 * A );
302  dydT1 = -B * myexp * invTime * (B/A-1) / ( T1 * T1);
303 
304  tempInvCovarianceMatrix[0] += dydT1 * dydT1; // (0,0)
305  tempInvCovarianceMatrix[1] += dydA * dydT1; // (0,1)
306  tempInvCovarianceMatrix[2] += dydB * dydT1; // (0,2)
307  tempInvCovarianceMatrix[3] += dydT1 * dydA; // (1,0)
308  tempInvCovarianceMatrix[4] += dydA * dydA; // (1,1)
309  tempInvCovarianceMatrix[5] += dydB * dydA; // (1,2)
310  tempInvCovarianceMatrix[6] += dydT1 * dydB; // (2,0)
311  tempInvCovarianceMatrix[7] += dydA * dydB; // (2,1)
312  tempInvCovarianceMatrix[8] += dydB * dydB; // (2,2)
313  }
314 
315  MeasureType SD = KWUtil::calcStandardDeviationArray<MeasureType>(nSamples, residuals);
316 
317  if (fabs(SD) < std::numeric_limits<MeasureType>::min())
318  return 1; //EXIT_FAILURE
319 
320  for (int i = 0; i < 3*3; ++i) {
321  tempInvCovarianceMatrix[i] = tempInvCovarianceMatrix[i] / (SD * SD);
322  }
323 
324  for (int i = 0; i < 3*3; ++i) {
325  invCovarianceMatrix[i] = tempInvCovarianceMatrix[i];
326  }
327 
328  return 0; // EXIT_SUCCESS
329  }
