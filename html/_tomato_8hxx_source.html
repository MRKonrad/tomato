60  readerPha->SetFileList(_opts->files_phase);
61  readerPha->Update();
62 
63  // have we read magnitudes?
64  if (readerMag->GetInvTimes().size() == 0){
65  throw std::runtime_error("No magnitude images read, check the paths!");
66  }
67 
68  typename SortInvTimesImageFilterType::Pointer sorterMag = SortInvTimesImageFilterType::New();
69  sorterMag->SetInvTimesNonSorted(readerMag->GetInvTimes());
70  sorterMag->SetInput(readerMag->GetOutput());
71  sorterMag->Update();
72 
73  typename SortInvTimesImageFilterType::Pointer sorterPha = SortInvTimesImageFilterType::New();
74  sorterPha->SetInvTimesNonSorted(readerPha->GetInvTimes());
75  sorterPha->SetInput(readerPha->GetOutput());
76  sorterPha->Update();
77 
78  // are the inversion times in magnitude and phase series equal?
79  if (sorterMag->GetInvTimesSorted() == sorterPha->GetInvTimesSorted()){
80  vnl_vector<InputPixelType > temp = sorterMag->GetInvTimesSorted();
81  _nSamples = temp.size();
82  _invTimes = new InputPixelType[_nSamples];
83  KWUtil::copyArrayToArray(_nSamples, _invTimes, temp.data_block());
84  } else {
85  throw std::runtime_error("Mag and Pha inv times are not equal");
86  }
87 
88  _dictionaryInput = readerMag->GetDicomIO()->GetMetaDataDictionary();
